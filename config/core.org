* Core

Core editor behavior, built-in settings, and the completion stack.
This module is loaded first and provides the foundation for everything else.

** Built-in Settings

Basic editor preferences: confirm with =y/n=, delete trailing whitespace on
save, disable lock files, auto-save, and backups. Tab key triggers completion
instead of indentation when point is mid-word.

#+BEGIN_SRC elisp
(defalias 'yes-or-no-p 'y-or-n-p)
(setenv "PAGER" "cat")

(use-package emacs
  :ensure nil
  :hook
  (before-save . delete-trailing-whitespace)
  :config
  (setq
   custom-safe-themes t
   kill-whole-line t
   case-fold-search nil
   ring-bell-function 'ignore
   create-lockfiles nil
   auto-save-default nil
   make-backup-files nil
   global-auto-revert-mode nil
   initial-major-mode 'org-mode
   column-number-mode t
   cursor-type 'box
   warning-minimum-level :emergency
   tab-always-indent 'complete
   display-time-default-load-average nil
   display-time-mail-string ""
   auth-sources "~/.authinfo"
   custom-file "~/.emacs.d/custom.el"))
#+END_SRC

** Scratch

Persistent, disposable scratch buffers. Scratches are unnamed by design
— they hold throwaway thoughts and quick notes. When killed (or on Emacs
exit), any scratch with content is saved to =~/.emacs-scratch/= with a
timestamped filename. Empty scratches are silently deleted, keeping the
buffer list tidy.

#+BEGIN_SRC elisp
(defvar my/scratch-dir "~/.emacs-scratch/"
  "Directory for persisted scratch buffers.")

(defun my/scratch--ensure-dir ()
  "Create the scratch directory if it doesn't exist."
  (make-directory my/scratch-dir t))

(defun my/scratch--filename ()
  "Return a timestamped filename for a scratch save."
  (expand-file-name (format-time-string "%Y%m%dT%H%M%S.org") my/scratch-dir))

(defun my/scratch--save (buffer)
  "Save scratch BUFFER to a timestamped file if it has content."
  (when (and (buffer-live-p buffer)
             (string-prefix-p "*scratch" (buffer-name buffer))
             (not (buffer-file-name buffer)))
    (with-current-buffer buffer
      (when (> (buffer-size) 0)
        (my/scratch--ensure-dir)
        (let ((filename (my/scratch--filename)))
          (write-region (point-min) (point-max) filename nil 'silent)
          (message "Scratch saved → %s" filename))))))

(defun my/scratch--cleanup-empty ()
  "Kill all empty scratch buffers (except the current one)."
  (dolist (buf (buffer-list))
    (when (and (string-prefix-p "*scratch" (buffer-name buf))
               (not (eq buf (current-buffer)))
               (buffer-live-p buf))
      (with-current-buffer buf
        (when (= (buffer-size) 0)
          (kill-buffer buf))))))

;;;###autoload
(defun my/scratch ()
  "Switch to the main *scratch* buffer, cleaning up empty scratches first.

If *scratch* doesn't exist, it is created in `org-mode' (or whatever
`initial-major-mode' is set to)."
  (interactive)
  (my/scratch--cleanup-empty)
  (let ((buf (get-buffer "*scratch*")))
    (unless buf
      (setq buf (get-buffer-create "*scratch*")))
    (switch-to-buffer buf))
  (when (fboundp 'org-mode)
    (unless (derived-mode-p 'org-mode)
      (org-mode))))

;;;###autoload
(defun my/scratch-new ()
  "Create a fresh scratch buffer with a unique name.

Empty scratches are cleaned up first, so only scratches with content
accumulate. The new buffer opens in `org-mode'."
  (interactive)
  (my/scratch--cleanup-empty)
  (let ((buf (generate-new-buffer-name "*scratch*")))
    (switch-to-buffer (get-buffer-create buf))
    (when (fboundp 'org-mode)
      (org-mode)))
  (message "New scratch buffer created"))

(defun my/scratch--save-on-kill (buffer &rest _)
  "Save scratch BUFFER before it is killed.

BUFFER may be a buffer object or a buffer name (string)."
  (when buffer
    (let ((buf (get-buffer buffer)))
      (when buf
        (my/scratch--save (or (buffer-base-buffer buf) buf))))))

(defun my/scratch--save-all ()
  "Save all live scratch buffers on Emacs exit."
  (dolist (buf (buffer-list))
    (my/scratch--save buf)))

;; Advice kill-buffer to persist scratches before they disappear
(advice-add 'kill-buffer :before #'my/scratch--save-on-kill)

;; Save all scratches when Emacs exits
(add-hook 'kill-emacs-hook #'my/scratch--save-all)
#+END_SRC

** Ghostel

Terminal emulator powered by [[https://github.com/dakra/ghostel][libghostty-vt]]. Skipped on Windows via the
=is-windows= check. The native module downloads automatically on first use, so
no =libvterm= system dependency is required. Shell integration (directory
tracking via OSC 7, prompt navigation) is on by default for bash, zsh and fish.
=hl-line-mode= is disabled inside ghostel for readability.

ghostel has no toggle package, so a few small helpers preserve the old
=vterm-toggle= / =vterm-toggle-cd= / buffer-cycling workflow.

#+BEGIN_SRC elisp
(use-package ghostel
  :unless is-windows
  :config
  (add-hook 'ghostel-mode-hook (lambda () (setq-local global-hl-line-mode nil)))

  (defun my/ghostel--buffers ()
    "Return live ghostel buffers in `buffer-list' order."
    (seq-filter (lambda (b) (eq (buffer-local-value 'major-mode b) 'ghostel-mode))
                (buffer-list)))

  (defun my/ghostel-toggle ()
    "Show or hide a ghostel terminal buffer (mimics `vterm-toggle')."
    (interactive)
    (let ((existing (my/ghostel--buffers)))
      (cond
       ((eq major-mode 'ghostel-mode) (bury-buffer))
       (existing (switch-to-buffer (car existing)))
       (t (ghostel)))))

  (defun my/ghostel-toggle-cd ()
    "Open or switch to ghostel at the current buffer's directory.
Mimics `vterm-toggle-cd'.  When creating a terminal it starts in the
current buffer's directory; shell integration keeps tracking afterwards."
    (interactive)
    (let ((dir (if-let ((file (buffer-file-name)))
                   (file-name-directory file)
                 default-directory)))
      (if (my/ghostel--buffers)
          (my/ghostel-toggle)
        (let ((default-directory dir))
          (ghostel)))))

  (defun my/ghostel-next ()
    "Switch to the next ghostel buffer (mimics `vterm-toggle-forward')."
    (interactive)
    (when-let ((next (cadr (memq (current-buffer) (my/ghostel--buffers)))))
      (switch-to-buffer next)))

  (defun my/ghostel-prev ()
    "Switch to the previous ghostel buffer (mimics `vterm-toggle-backward')."
    (interactive)
    (when-let ((prev (cadr (memq (current-buffer) (reverse (my/ghostel--buffers))))))
      (switch-to-buffer prev)))

  ;; preserve muscle memory from the old vterm-toggle setup
  (global-set-key (kbd "s-t")   #'my/ghostel-toggle)
  (global-set-key (kbd "C-M-t") #'my/ghostel-toggle-cd)
  (with-eval-after-load 'ghostel
    (define-key ghostel-semi-char-mode-map (kbd "C-c t") #'ghostel-copy-mode)
    (define-key ghostel-semi-char-mode-map (kbd "s-n")   #'my/ghostel-next)
    (define-key ghostel-semi-char-mode-map (kbd "s-p")   #'my/ghostel-prev)))
#+END_SRC

** Completion Stack

Vertico, Corfu, Orderless, Consult, Cape, and Tempel work together to
provide completion everywhere: minibuffer, in-buffer, and templates.

*** Vertico

Vertical minibuffer completion. Enabled globally.

#+BEGIN_SRC elisp
(use-package vertico
  :init
  (vertico-mode))
#+END_SRC

*** Tempel

Template expansion using =tempel-collection= for ready-made snippets.
Hooked into prog, conf, and text modes so templates are available
everywhere.

#+BEGIN_SRC elisp
(use-package tempel
  :bind (("M-+" . tempel-complete)
         ("M-*" . tempel-insert))
  :init
  (defun tempel-setup-capf ()
    (setq-local completion-at-point-functions
                (cons #'tempel-expand completion-at-point-functions)))
  (add-hook 'conf-mode-hook 'tempel-setup-capf)
  (add-hook 'prog-mode-hook 'tempel-setup-capf)
  (add-hook 'text-mode-hook 'tempel-setup-capf))

(use-package tempel-collection)
#+END_SRC

*** Corfu

In-buffer popup completion. Auto-triggers as you type and dismisses when
no match remains after a separator.

#+BEGIN_SRC elisp
(use-package corfu
  :custom
  (corfu-auto t)
  (corfu-quit-no-match 'separator)
  :init
  (global-corfu-mode))
#+END_SRC

*** Orderless

Space-separated component matching. The component separator is =[ &]= so
spaces and ampersands split search terms.

#+BEGIN_SRC elisp
(use-package orderless
  :custom
  (completion-styles '(orderless basic))
  (orderless-component-separator "[ &]")
  (completion-category-overrides '((file (styles basic partial-completion)))))
#+END_SRC

*** Consult

Search and switch commands. =consult-line-literal= replaces isearch with
substring-based line search.

#+BEGIN_SRC elisp
(use-package consult
  :bind
  ("C-x b" . consult-buffer)
  ("C-c s" . consult-ripgrep)
  ("C-c h" . consult-history)
  ("C-c k" . consult-kmacro)
  ("C-c i" . consult-info)
  ("C-s" . consult-line-literal)
  :init
  (setq register-preview-delay 0.5
        register-preview-function #'consult-register-format))

(defun consult-line-literal ()
  "Use this instead of isearch."
  (interactive)
  (let ((completion-styles '(substring))
        (completion-category-defaults nil)
        (completion-category-overrides nil))
    (consult-line)))
#+END_SRC
